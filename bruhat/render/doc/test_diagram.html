from bruhat.render.box import HBox
from bruhat.render.diagram import HDia, VDia, VWire, Cap, Cup, SIZE

top = HDia([VWire(), Cap()])
bot = HDia([Cup(), VWire()])
lsnake = VDia([top, bot])

bot = HDia([VWire(), Cup()])
rsnake = VDia([top, bot])

dia.render(cvs)

Instead of explicitly using VDia and HDia we can use the operators * and @, respectively. The * operator composes top-down, like VBox. The @ operator composes left to right, like HBox.

from bruhat.render.box import Box
from bruhat.render.diagram import Spider, VWire, Cap, Cup
Box.DEBUG = True

box = Spider(2, 2) @ VWire()
box = box * (Spider(3, 1))
box = box @ VWire()
box = box * Spider(2, 0, weight=0.9)
box = (Cap() @ Spider(0, 1) @ Spider(0, 1)) * box

from bruhat.render.box import Box, HBox
from bruhat.render.diagram import VWire, Braid
Box.DEBUG = False

Id = VWire

scale = 2.0
w = 1.5*scale
h = 0.5*scale
s12 = lambda : Braid(min_width=w, min_height=h) @ Id(min_height=h, min_width=h)
s23 = lambda : Id(min_height=h, min_width=h) @ Braid(min_width=w, min_height=h)

lhs = s12() * s23() * s12()
rhs = s23() * s12() * s23()
box = HBox([lhs, "$=$", rhs], align="center")


from random import shuffle, seed, randint
from operator import matmul
from functools import reduce

from bruhat.render import canvas, color, style, path
from bruhat.render.box import Box, HBox
from bruhat.render.diagram import VWire, Braid, Relation

seed(1)

scale = 0.3
w = 1.4*scale
h = 1.8*scale
Id = lambda : VWire(min_height=scale, min_width=scale)
Swap = lambda inverse : Braid(inverse=inverse, min_width=w, min_height=h, space=0.5)

box = None
m, n = 3, 3
k = 2*m+n
for count in range(3):
    items = [Swap(randint(0,1)) for k in range(m)] + [Id() for k in range(n)]
    shuffle(items)
    row = reduce(matmul, items)
    if box is None:
        box = row
    else:
        box = box * row

Now we take the closure of this braid:

lhs = reduce(matmul, [Id() for i in range(k)])
box = lhs @ box
rels = [(i, 2*k-i-1) for i in range(k)]
rel = Relation(0, 2*k, botbot=rels, weight=200.0)
box = rel * box
rels = [(i, 2*k-i-1) for i in range(k)]
rel = Relation(2*k, 0, toptop=rels, weight=200.0)
box = box * rel

cvs = canvas.canvas()

system = box.layout(cvs)
